jQuery Schedule Demo v2.0

Basic

$("#schedule").jqs();

Read mode

$("#schedule2").jqs({
    mode: "read",
    data: [{
        day: 0,
        periods: [
            ["20:00", "00:00"],
            ["20:00", "22:00"], // Invalid period, not displayed
            ["00:00", "02:00"]
        ]
    }, {
        day: 3,
        periods: [
            ["00:00", "08:30"],
            ["09:00", "12:00"]
        ]
    }]
});

Events & Methods

$("#schedule3").jqs({
    onInit: function () {
        $("#log").val("onInit fire !\n" + $("#log").val());
    },
    onAddPeriod: function () {
        $("#log").val("onAddPeriod fire !\n" + $("#log").val());
    },
    onRemovePeriod: function () {
        $("#log").val("onRemovePeriod fire !\n" + $("#log").val());
    },
    onPeriodClicked: function () {
        $("#log").val("onPeriodClicked fire !\n" + $("#log").val());
    }
});

$("#schedule3").jqs('export');

$("#schedule3").jqs('import', [
    {
        day: 1,
        periods: [
            ["8:00", "10:00"]
        ]
    }, {
        day: 4,
        periods: [
            ["20:00", "00:00"],
            ["20:00", "22:00"], // Invalid period, not displayed
        ]
    }
]);

$("#schedule3").jqs('reset');

Translation & 12-hour clock

$("#schedule4").jqs({
    mode: "read",
    days: ["Lundi", "Mardi", "Mercredi", "Jeudi", "Vendredi", "Samedi", "Dimanche"],
    hour: 12,
    data: [
        {
            day: 0,
            periods: [
                ["10pm", "12am"],
                ["12am", "2:30am"]
            ]
        }, {
            day: 1,
            periods: [
                ["1:45am", "5:15am"],
                ["10am", "12pm"]
            ]
        }
    ]
});
