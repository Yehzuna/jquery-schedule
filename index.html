jQuery Schedule Demo

Edition mode

$("#schedule").jqs();

$("#schedule").jqs('export');

$("#schedule").jqs('import', [
     {
         day: 1,
         periods: [
             ["8:00", "10:00"],
         ]
     }, {
        day: 4,
        periods: [
            ["20:00", "00:00"],
            ["00:00", "02:00"]
        ]
    }
]);

$("#schedule").jqs('reset');

Read mode

$("#schedule2").jqs({
    mode: "read",
    data: [
        {
            day: 0,
            periods: [
                ["20:00", "00:00"],
                ["00:00", "02:00"]
            ]
        }, {
            day: 3,
            periods: [
                ["00:00", "08:30"],
                ["09:00", "12:00"]
            ]
        }
    ]
});

Translation & 12-hour clock

$("#schedule3").jqs({
    mode: "read",
    days: ["Lundi", "Mardi", "Mercredi", "Jeudi", "Vendredi", "Samedi", "Dimanche"],
    hour: 12,
    data: [
        {
            day: 0,
            periods: [
                ["10pm", "12am"],
                ["12am", "2:30am"]
            ]
        }
    ]
});
